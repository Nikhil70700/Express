<!-- 
1. Query parameters are key-value pairs appended to a URL after a ?, separated by & ,like /search?query=express&limit=10 .

-->

<!-- 
2. In Express.js ,they can be accessed using req.query, which returns an abject containing the parameters.
 -->


<!-- 
 3. For example ,in /search?page=4,req.query.page will give "page".
  -->

<!-- 
  4. Query parameters are often used to pass optional or filter data to the server without modifying the route.
-->

<!-- 
5. They are always part of the URL and visible in the browser address bar.
